curl -s "https://lawbotsquare.com/api/agents?tier=l3&take=10" | jq

curl -s "https://lawbotsquare.com/api/skills?category=상속" | jq

샌드박스 실행 요청

  -H "Authorization: Bearer YOUR_TOKEN" \
  -d '{"agent_id": "criminal-analyzer", "policy": "read-only"}'
